# Indefinite Integrals of $trig(x)$

[Index](trig_inverse_trig_int_ref.md)

## sin

$$
\begin{aligned}
    &\dfrac{\mathrm{d}}{\mathrm{d}x}\cos(x) = -\sin(x) \\
    &\therefore \dfrac{\mathrm{d}}{\mathrm{d}x}(-\cos(x)) = \sin(x) \\
    &\therefore \int \sin(x) \,\mathrm{d}x = -\cos(x) + C
\end{aligned}
$$

## cos

$$
\begin{aligned}
    &\dfrac{\mathrm{d}}{\mathrm{d}x}\sin(x) = \cos(x) \\
    &\therefore \int \cos(x) \,\mathrm{d}x = \sin(x) + C
\end{aligned}
$$

## tan

$$
\begin{aligned}
    \int \tan(x) \,\mathrm{d}x &= \int \dfrac{\sin(x)}{\cos(x)} \,\mathrm{d}x \\
                               &= \int \dfrac{1}{\cos(x)} \,\mathrm{d}(-\cos(x)) \\
                               &= -\int \dfrac{1}{\cos(x)} \,\mathrm{d}(\cos(x)) \\
                               &= -\ln(\vert \cos(x) \vert) + C \\
                               &\overset{\text{or}}{=} \ln(\vert \sec(x) \vert) + C
\end{aligned}
$$

## cot

$$
\begin{aligned}
    \int \cot(x) \,\mathrm{d}x &= \int \dfrac{\cos(x)}{\sin(x)} \,\mathrm{d}x \\
                               &= \int \dfrac{1}{\sin(x)} \,\mathrm{d}(\sin(x)) \\
                               &= \ln(\vert \sin(x) \vert) + C \\
                               &\overset{\text{or}}{=} -\ln(\vert \csc(x) \vert) + C
\end{aligned}
$$

## sec

$$
\begin{aligned}
    \dfrac{\mathrm{d}}{\mathrm{d}x} \sec(x) &= \tan(x)\sec(x) \\
    \dfrac{\mathrm{d}}{\mathrm{d}x} \tan(x) &= \sec^2(x) \\
    \therefore 
    \dfrac{\mathrm{d}}{\mathrm{d}x} (\sec(x) + \tan(x)) &= \sec(x)(\tan(x) + \sec(x))\\
\\
\therefore
    \int \sec(x) \,\mathrm{d}x &= \int \sec(x) \dfrac{\tan(x) + \sec(x)}{\tan(x) + \sec(x)} \,\mathrm{d}x \\
                               &= \int \dfrac{\sec(x)(\tan(x) + \sec(x))}{\tan(x) + \sec(x)} \,\mathrm{d}x \\
                               &= \int \dfrac{1}{\tan(x) + \sec(x)} \, \mathrm{d}(\sec(x) + \tan(x))\\
                               &\overset{u=\tan(x)+\sec(x)}{=} \int \dfrac{1}{u} \, \mathrm{d}u\\
                               &= \ln(\vert u \vert) + C \\
                               &= \ln(\vert \tan(x)+\sec(x) \vert) + C
\end{aligned}
$$

## csc

$$
\begin{aligned}
    \dfrac{\mathrm{d}}{\mathrm{d}x} \csc(x) &= -\cot(x)\csc(x) \\
    \dfrac{\mathrm{d}}{\mathrm{d}x} \cot(x) &= -\csc^2(x) \\
    \therefore 
    \dfrac{\mathrm{d}}{\mathrm{d}x} (\csc(x) + \cot(x)) &= -\csc(x)(\cot(x) + \csc(x))\\
\\
\therefore
    \int \csc(x) \,\mathrm{d}x &= \int \csc(x) \dfrac{\cot(x) + \csc(x)}{\cot(x) + \csc(x)} \,\mathrm{d}x \\
                               &= \int \dfrac{\csc(x)(\cot(x) + \csc(x))}{\cot(x) + \csc(x)} \,\mathrm{d}x \\
                               &= -\int \dfrac{1}{\cot(x) + \csc(x)} \, \mathrm{d}(\cot(x) + \csc(x))\\
                               &\overset{u=\cot(x)+\csc(x)}{=} -\int \dfrac{1}{u} \, \mathrm{d}u\\
                               &= -\ln(\vert u \vert) + C \\
                               &= -\ln(\vert \cot(x)+\csc(x) \vert) + C
\end{aligned}
$$
